\section*{Greek Letters and Statistical Symbols}

\begin{description}
	
	\item[$\alpha$] Significance level (probability of a Type I error)
	
	\item[$\alpha_{\text{base}}$] Baseline significance level predicted by the machine learning model
	
	\item[$\alpha_{\text{final}}$] Final recommended significance level after context adjustments
	
	\item[$\alpha_{\text{opt}}$] Optimal significance level derived from replication data
	
	\item[$\beta$] Probability of a Type II error
	
	\item[$\eta^2$] Eta-squared; effect size measure for F-tests
	
	\item[$\phi$] Phi coefficient; effect size for chi-squared tests
	
	\item[$\mu$] Population mean
	
	\item[$\sigma$] Population standard deviation
	
	\item[$\chi^2$] Chi-squared test statistic
	
\end{description}

\section*{Roman Letters and Mathematical Notation}

\begin{description}
	
	\item[$A$] Candidate set of significance levels used for optimal alpha selection
	
	\item[$d$] Standardised effect size (Cohen's $d$)
	
	\item[$df$] Degrees of freedom
	
	\item[$df_1$] Numerator degrees of freedom (F-test)
	
	\item[$df_2$] Denominator degrees of freedom (F-test)
	
	\item[$F$] F-test statistic
	
	\item[$H_0$] Null hypothesis
	
	\item[$H_1$] Alternative hypothesis
	
	\item[$k_1$] Risk tolerance adjustment factor
	
	\item[$k_2$] Test directionality adjustment factor
	
	\item[$k_3$] Multiple testing correction factor
	
	\item[$L$] Expected loss function
	
	\item[$m$] Number of simultaneous hypothesis tests
	
	\item[$N$] Sample size
	
	\item[$n$] Number of observations in the test set
	
	\item[$p$] p-value
	
	\item[$p_{\text{orig}}$] p-value reported in the original study
	
	\item[$r$] Pearson correlation coefficient (effect size)
	
	\item[$t$] t-test statistic
	
	\item[$T$] General test statistic
	
	\item[$t_{\text{obs}}$] Observed test statistic value
	
	\item[$Z$] Standardised test statistic (z-score)
	
\end{description}

\section*{Constants and Fixed Thresholds}

\begin{description}
	
	\item[0.05] Conventional significance level
	
	\item[0.005] Proposed stricter significance threshold
	
	\item[0.001] Lower bound applied to $\alpha_{\text{final}}$
	
	\item[0.15] Upper bound applied to $\alpha_{\text{final}}$
	
	\item[42] Random seed used for reproducible model training
	
	\item[200] Number of decision trees in the Random Forest model
	
	\item[0.80 / 0.20] Training / test split ratio used during model evaluation
	
\end{description}
