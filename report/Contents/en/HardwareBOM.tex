\subsection{Specifications of Development Environment}

\textbf{Developement Enviornment:}
\begin{enumerate}
	\item \textbf{Operating system}:
	\begin{itemize}
		\item Windows\texttrademark 11 (64 bit) version 23H2 (OS Build 22631.3737)
	\end{itemize}
	
	\item \textbf{Processors}:
	\begin{itemize}
		\item Intel Core i7-4710HQ processor (2.5GHz quad-core with Turbo Boost up
		to 3.5GHz)
	\end{itemize}
	
	\item \textbf{RAM}:
	\begin{itemize}
		\item 16GB DDR3L 1600 MHz SDRAM
	\end{itemize}
	
	\item \textbf{Storage}:
	\begin{itemize}
		\item 512 GB SSD. A minimum of 100GB of free disk space is recommended.
	\end{itemize}
	
	\item \textbf{GPU}:
	\begin{itemize}
		\item NVIDIA GeForce GTX 860M with 4GB GDDR5 VRAM
	\end{itemize}
\end{enumerate}
