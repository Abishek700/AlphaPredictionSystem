\section{Specifications}

\begin{itemize}
	\item \textbf{Operating Voltage:} 3.3V
	\item \textbf{Power Consumption:}
	\begin{itemize}
		\item 15mA in low-power mode
		\item Up to 60mA in active mode
	\end{itemize}
	\item \textbf{Operating Temperature Range:} -40°C to 85°C
	\item \textbf{Dimensions:} 45mm x 18mm
	\item \textbf{Weight:} 20 grams
	\item \textbf{Battery:} 2 x CR2032 (Replace every 4 months)
	\item \textbf{Sensors:}
	\begin{itemize}
		\item 9-axis IMU (LSM9DS1) with Accelerometer, Gyroscope, Magnetometer
	\end{itemize}
	\item \textbf{LED Indicators:}
	\begin{itemize}
		\item Blinking Blue (Normal gait)
		\item Blinking Red (Abnormal gait)
		\item Constant Black (Data error)
		\item Constant Orange (Low battery level)
	\end{itemize}
