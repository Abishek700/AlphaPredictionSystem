\section{Key Features}

\textbf{Device Highlights:}

\begin{enumerate}
	\item \textbf{Compact Size}: The gait pattern device has a case size of 72mm x 53mm.
	\item \textbf{No Internet or Bluetooth Needed}: The device operates autonomously, providing visual feedback via LEDs.
	\item \textbf{Power Efficient}: Low power consumption  runtime of 4 hours in active mode.
	\item \textbf{Accurate Gait Detection}: Utilizes a 9-axis IMU to analyze walking patterns and detect abnormalities.
	\item \textbf{Visual Alerts:}
	
	\begin{itemize}
		\item \textbf{Blue LED}: Indicates normal gait.
		
		
		\item \textbf{Red LED}: Indicates abnormal gait.
	
		
		
		\item \textbf{Black LED}: Signals a data error.
		
		
		
		\item \textbf{Yellow LED Blinking }:  Indicates a low battery level.
		
	\end{itemize}
	\item \textbf{Durable Design}: Operates in a temperature range of -40°C to 85°C.
\end{enumerate}
