\subsection{How to Replace the Battery}

\begin{enumerate}
	\item \textbf{Switch Off the Device:} Ensure the device is off, and verify that no lights are illuminated on the board.
	\begin{center}
		\includegraphics[width=0.8\textwidth]{Images/SwitchON.jpg} % Adjust the path as needed
	\end{center}
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/TopperspectiveON.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Unscrew the Cover:} Remove the 4 screws and nuts holding the cover and body of the device.
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/UnscrewscrewOFF.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Remove the Cover:} Carefully remove the cover with the applied button.
	\begin{center}
		\includegraphics[width=1.1\textwidth]{Images/RemovetransparentON.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Remove the Board:} Use the provided hand sticks to carefully lift the board out. Be cautious, as the battery wires are approximately 10 cm long.
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/removeboardoff.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Open the Battery Compartment:} Open the battery case and remove the old two batteries.
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/baterycase.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Insert New Batteries:}
	\begin{itemize}
		\item \textbf{Battery Closer to the Wire:} Insert with the plus (+) symbol facing up.
		\item \textbf{Second Battery:} Insert with the minus (-) symbol facing up.
		\item Follow the picture instructions for proper placement.
	\end{itemize}
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/batterySRON.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Reposition the Board:} Place the board back in its original position as shown in the reference photo.
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/removeboardON.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Reattach the Cover:} Ensure the cover is correctly aligned, with the button on the opposite side of the USB port on the dashboard.
	\begin{center}
		\includegraphics[width=1.2\textwidth]{Images/RemovetransparentOFF.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Secure with Screws and Nuts:} Reinsert the 4 screws and secure them with the nuts.
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/UnscrewscrewON.jpg} % Adjust the path as needed
	\end{center}
	
	\item \textbf{Test the System:} Turn the device ON and verify that the green LED on the left side of the USB port is illuminated, indicating the device is powered and operational.
	\begin{center}
		\includegraphics[width=1\textwidth]{Images/TopperspectiveON.jpg} % Adjust the path as needed
	\end{center}
	
	
	
\end{enumerate}
