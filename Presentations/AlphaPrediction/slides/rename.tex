

\section{Problem}

\frame{\frametitle{Hypothesis Testing}
	
	\begin{itemize}
		\item Decisions in hypothesis testing are based on a simple rule:
		
		\[
		\begin{aligned}
			\text{p-value} &\;\longrightarrow\; \text{Compare with } \alpha \\
			p \leq \alpha &\;\Rightarrow\; \text{Reject } H_0 \\
			p > \alpha &\;\Rightarrow\; \text{Fail to reject } H_0
		\end{aligned}
		\]
		
		\item The choice of $\alpha$ directly affects the decision.
		
	\end{itemize}
	
	\footnotesize \cite{Fisher1925}
	
}
\frame{\frametitle{Why do we use $\alpha = 0.05$?}
	
	\begin{itemize}
		\item $\alpha$ = probability of a Type I error
		\item $\alpha = 0.05$ is commonly used as a standard threshold
		\item This choice is largely based on historical convention
		\item There is no clear justification for its universal use
	\end{itemize}
	
	\footnotesize \cite{Fisher1925,CowlesDavis1982}
	
}

\frame{\frametitle{The Problem with $\alpha = 0.05$}
	
	\begin{columns}[T]
		
		\column{0.33\textwidth}
		\begin{block}{One Size Fits All}
			\begin{minipage}[t][3.5cm][t]{\linewidth}
				Same $\alpha$ used regardless of
				\begin{itemize}
					\item sample size
					\item effect size
				\end{itemize}
			\end{minipage}
		\end{block}
		
		\column{0.33\textwidth}
		\begin{block}{Cliff-Edge Effect}
			\begin{minipage}[t][3.5cm][t]{\linewidth}
				\begin{itemize}
					\item $p = 0.049$ $\rightarrow$ Significant
					\item $p = 0.051$ $\rightarrow$ Not significant
				\end{itemize}
			\end{minipage}
		\end{block}
		
		\column{0.33\textwidth}
		\begin{block}{Replication Crisis}
			\begin{minipage}[t][3.5cm][t]{\linewidth}
				\begin{itemize}
					\item Many statistically significant findings fail to replicate.
				\end{itemize}
			\end{minipage}
		\end{block}
		
	\end{columns}
	
	
	
	\footnotesize \cite{OpenScience2015,WassersteinLazar2016}
	
}

\section{Existing approaches}

\frame{\frametitle{Existing Approaches for Choosing $\alpha$}
	
\begin{itemize}
	\item Fixed $\alpha$ levels ($0.10,\; 0.05,\; 0.01$)
	\item Power-based approaches (sample size, effect size)
	\item Decision-theoretic approaches (error costs)
	\item Stricter thresholds ($\alpha = 0.005$)
\end{itemize}

\vspace{0.01cm}

\begin{center}
	$\Longrightarrow$ \; \textbf{All approaches still rely on a fixed $\alpha$}
\end{center}
	
	\footnotesize \cite{Cohen1988,Berger1985,Benjamin2018}
	
}

\frame{\frametitle{Limitations of Current Approaches}
	
	\begin{itemize}
		\item Still rely on a fixed $\alpha$
		\item No adaptation to study characteristics
		\item No practical system for selecting $\alpha$
	\end{itemize}
	
	\vspace{0.1cm}
	
	\begin{center}
		$\Longrightarrow$ \; \textbf{This motivates a data driven approach}
	\end{center}
	
}

\begin{frame}{Optimal Alpha Definition}
	
	\begin{columns}[T]
		
		\column{0.48\textwidth}
		\begin{block}{Replicated}
			Keep original $\alpha$  
			
			(decision validated)
		\end{block}
		
		\column{0.48\textwidth}

	\end{columns}

\section{Stage 2}
\begin{frame}{Stage 2: Context-Based Adjustment}
	
	\begin{block}{Reason}
		
	Baseline $\alpha$ is not sufficient
		
	\end{block}
	
	\vspace{0.4cm}
	
	\begin{exampleblock}{Adjustment Factors}
		
		\begin{itemize}
			\item Risk tolerance
			\item Test direction (one-tailed vs two-tailed)
			\item Number of hypothesis tests (multiple testing)
		\end{itemize}
		
	\end{exampleblock}
	
\end{frame}

\begin{frame}{Adjustment Factors}
	
	\begin{columns}[T]
		
		\column{0.32\textwidth}
		\begin{block}{$k_1$ — Risk Tolerance}
			
			\begin{itemize}
				\item Very Strict $\times 0.50$
				\item Strict $\times 0.75$
				\item Neutral $\times 1.00$
				\item Lenient $\times 1.25$
				\item Very Lenient $\times 1.50$
			\end{itemize}
			
		\end{block}
		
		
		\column{0.32\textwidth}
		\begin{block}{$k_2$ — Test Direction}
			
			\begin{itemize}
				\item Two-tailed $\times 1.00$
				\item One-tailed $\times 2.00$
			\end{itemize}
			
			\vspace{0.3cm}
			
			Reflects rejection region allocation.
			
		\end{block}
		
		
		\column{0.32\textwidth}
		\begin{block}{$k_3$ — Multiple Testing}
			
			Bonferroni correction:
			
			\[
			\alpha / m
			\]
			
			where $m$ = number of hypothesis tests.
			
		\end{block}
		
	\end{columns}
	
	\vspace{0.4cm}
	
	\centering
	\small
	Final clamping: $0.001 \leq \alpha_{final} \leq 0.15$
	
\end{frame}
\begin{frame}{Final Alpha Calculation}
	
	\begin{block}{Adjusted Significance Level}
		
		\[
		\alpha_{final}
		=
		\alpha_{baseline}
		\times
		k_1
		\times
		k_2
		\times
		k_3
		\]
		
	\end{block}
	
	\vspace{0.5cm}
	
	\begin{exampleblock}{Interpretation}
		
		\begin{itemize}
			\item $\alpha_{baseline}$: predicted by the machine learning model.
			\item $k_1$: adjustment for risk tolerance.
			\item $k_2$: adjustment for test direction.
			\item $k_3$: adjustment for multiple hypothesis testing.
		\end{itemize}
		
	\end{exampleblock}
	
\end{frame}
