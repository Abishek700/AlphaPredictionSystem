

\section{What is Alpha in Hypothesis Testing}

\frame{\frametitle{Hypothesis Testing}
	
	\textbf{Observed Result}
	\begin{itemize}
		\item “A method shows improvement”
	\end{itemize}
	
	\vspace{0.2cm}
	
	\textbf{Key Question}
	\begin{center}
		\Large
		Is it \textbf{real} or just \textbf{coincidence}?
	\end{center}
	
	
	\begin{center}
		$\Downarrow$
	\end{center}
	
	\textbf{Hypothesis Testing Procedure}
	\begin{itemize}
		\item Assume $H_0$: No effect  
		\item Choose $\alpha$ (risk level)  
		\item Compute p-value (under $H_0$)  
		\item Compare p-value with $\alpha$  
	\end{itemize}
	
}
\frame{\frametitle{Decision Rule}
	
	\begin{center}
		\Large
		Decision based on p-value and $\alpha$
	\end{center}
	
	\vspace{0.6cm}
	
	\begin{center}
		\begin{tabular}{c}
			$p \leq \alpha \;\Rightarrow\;$ Reject $H_0$ \\
			\vspace{0.3cm} \\
			$p > \alpha \;\Rightarrow\;$ Do not reject $H_0$
		\end{tabular}
	\end{center}
	
	\vspace{0.8cm}
	
	\begin{center}
		\alert{\Large $\alpha$ determines what is considered ``surprising''}
	\end{center}
	
}
\section{Problem}

\frame{\frametitle{The Problem with $\alpha = 0.05$}
	
	\begin{itemize}
	\item $\alpha = 0.05$ become standard threshold with no clear justification
\end{itemize}
	\begin{columns}[T]
		
		\column{0.33\textwidth}
		\begin{block}{One Size Fits All}
			\begin{minipage}[t][3 cm][t]{\linewidth}
				Same $\alpha$ used regardless of
				\begin{itemize}
					\item sample size
					\item effect size
				\end{itemize}
			\end{minipage}
		\end{block}
		
		\column{0.33\textwidth}
		\begin{block}{Cliff-Edge Effect}
			\begin{minipage}[t][3cm][t]{\linewidth}
				\begin{itemize}
					\item $p = 0.049$ $\rightarrow$ Significant
					\item $p = 0.051$ $\rightarrow$ Not significant
				\end{itemize}
			\end{minipage}
		\end{block}
		
		\column{0.33\textwidth}
		\begin{block}{Replication Crisis}
			\begin{minipage}[t][3cm][t]{\linewidth}
				\begin{itemize}
					\item Many statistically significant findings fail to replicate.
				\end{itemize}
			\end{minipage}
		\end{block}
		
	\end{columns}
	
	
	
	\footnotesize \cite{OpenScience2015,WassersteinLazar2016}
	
}

\section{Existing approaches}

\frame{\frametitle{Existing Approaches for Choosing $\alpha$}
	
\begin{itemize}
	\item Fixed $\alpha$ levels ($0.10,\; 0.05,\; 0.01$)
	\item Power-based approaches (sample size, effect size)
	\item Decision-theoretic approaches (error costs)
	\item Stricter thresholds ($\alpha = 0.005$)
\end{itemize}

\vspace{0.01cm}

\begin{center}
	$\Longrightarrow$ \; \textbf{All approaches still rely on a fixed $\alpha$}
\end{center}
	
	\footnotesize \cite{Cohen1988,Berger1985,Benjamin2018}
	
}

\begin{frame}{Optimal Alpha Definition}
	
	\begin{columns}[T]
		
		\column{0.48\textwidth}
		\begin{block}{Replicated}
			Keep original $\alpha$  
			
			(decision validated)
		\end{block}
		
		\column{0.48\textwidth}
		\begin{alertblock}{Failed}
			Adjust $\alpha$: largest  
			
			$\alpha < p$-value  
			
			(correct decision)
		\end{alertblock}
		
	\end{columns}
	
	\vspace{0.5cm}
	
	\begin{exampleblock}{Example}
		
		$p = 0.047 \rightarrow$ failed to replicate  
		
		Optimal $\alpha = 0.04$ (not 0.05)
		
	\end{exampleblock}
	

\end{frame}

\section{Stage 2}
\begin{frame}{Stage 2: Context Based Adjustment}
	
	\begin{block}
		
	Baseline $\alpha$ is not sufficient
		
	\end{block}
	
	\vspace{0.4cm}
	
	\begin{exampleblock}{Reason}
		
		\begin{itemize}
			\item Same inputs can have different study contexts
			\item Baseline $\alpha$ does not reflect user preferences or risk
			\item Real world decisions require context aware adjustment
		\end{itemize}
		
	\end{exampleblock}
	
\end{frame}

\begin{frame}{Adjustment factors}
	
	\begin{columns}[T]
		
		\column{0.32\textwidth}
		\begin{block}{$k_1$ — Risk Tolerance}
			
			\begin{itemize}
				\item Very Strict $\times 0.50$
				\item Strict $\times 0.75$
				\item Neutral $\times 1.00$
				\item Lenient $\times 1.25$
				\item Very Lenient $\times 1.50$
			\end{itemize}
			
		\end{block}
		
		
		\column{0.32\textwidth}
		\begin{block}{$k_2$ — Test Direction}
			
			\begin{itemize}
				\item Two-tailed $\times 1.00$
				\item One-tailed $\times 2.00$
			\end{itemize}
			
		\end{block}
		
		
		\column{0.32\textwidth}
		\begin{block}{$k_3$ — Multiple Testing}
			
			Bonferroni:
			
			\[
			\alpha / m
			\]
			
		\end{block}
		
	\end{columns}
	
	\vspace{0.4cm}
	
	\centering
	
	\textbf{
		$\alpha_{final}
		=
		\alpha_{baseline}
		\times
		k_1
		\times
		k_2
		\times
		k_3$
	}
	
	\vspace{0.2cm}
	
	\small
	Final range: $0.001 \leq \alpha_{final} \leq 0.15$
	
\end{frame}

\section{Conclusion}
\begin{frame}{Conclusion}
	
	\begin{block}{Key Takeaways}
		\begin{itemize}
			\item $\alpha = 0.05$ is not universally appropriate
			\item $\alpha$ should depend on study characteristics
			\item A data-driven approach improves $\alpha$ selection
			\item Combines machine learning with interpretable adjustments
			\item \textbf{Statistical significance should be adaptive, not fixed}
		\end{itemize}
	\end{block}
	
\end{frame}
