

\section{Motivation}

\frame{\frametitle{Hypothesis Testing}
	
	\begin{itemize}
		\item Statistical hypothesis testing is widely used in empirical research.
		\item Researchers test whether an observed effect reflects a real relationship or random variation.
		
	\end{itemize}
	\vspace{0.6cm}
	
	\begin{center}
		\Large
		p-value $\;\longrightarrow\;$ Compare with $\alpha$
		\\[0.4cm]
		
		\begin{tabular}{c}
			$p \leq \alpha$ \quad $\Rightarrow$ Reject $H_0$ \\
			$p > \alpha$ \quad $\Rightarrow$ Fail to Reject $H_0$
		\end{tabular}
		
	\end{center}
	\footnotesize \cite{Fisher1925}
	
}

\frame{\frametitle{Why do we use $\alpha = 0.05$?}
	
	\begin{itemize}
		\item $\alpha$ represents the probability of a Type I error.
		\item In practice, researchers often use $\alpha = 0.05$ as a standard threshold.
		\item This value became widely adopted as a historical convention.
	\end{itemize}
	
	\footnotesize \cite{Fisher1925,CowlesDavis1982}
	
}

\frame{\frametitle{The Problem with $\alpha = 0.05$}
	
	\begin{columns}[T]
		
		\column{0.33\textwidth}
		\begin{block}{One Size Fits All}
			\begin{minipage}[t][3.5cm][t]{\linewidth}
				Same threshold applied regardless of
				\begin{itemize}
					\item sample size
					\item effect size
				\end{itemize}
			\end{minipage}
		\end{block}
		
		\column{0.33\textwidth}
		\begin{block}{Cliff-Edge Effect}
			\begin{minipage}[t][3.5cm][t]{\linewidth}
				\begin{itemize}
					\item $p = 0.049$ $\rightarrow$ Significant
					\item $p = 0.051$ $\rightarrow$ Not significant
				\end{itemize}
			\end{minipage}
		\end{block}
		
		\column{0.33\textwidth}
		\begin{block}{Replication Crisis}
			\begin{minipage}[t][3.5cm][t]{\linewidth}
				\begin{itemize}
					\item Many statistically significant findings fail to replicate.
				\end{itemize}
			\end{minipage}
		\end{block}
		
	\end{columns}
	
	
	
	\footnotesize \cite{OpenScience2015,WassersteinLazar2016}
	
}

\section{Existing Approaches}

\frame{\frametitle{Existing Approaches for Choosing $\alpha$}
	
	\begin{itemize}
		\item Fixed significance levels (0.10, 0.05, 0.01).
		\item Power-based approaches considering sample size and effect size.
		\item Decision-theoretic approaches based on error costs.
		\item Recent proposals suggest stricter thresholds such as $\alpha = 0.005$.
	\end{itemize}
	
	\footnotesize \cite{Cohen1988,Berger1985,Benjamin2018}
	
}
\frame{\frametitle{Limitations of Existing Approaches}
	
	\begin{itemize}
		\item Most approaches still rely on fixed thresholds.
		\item They do not adapt the significance level to study characteristics.
		\item There is no widely used practical system for selecting $\alpha$.
	\end{itemize}
	
}

\section{Research Objective}

\frame{\frametitle{Research Objective}
	
	\begin{itemize}
		\item Build a system that recommends a Study specific  ($\alpha$).
		\item The recommendation is based on study or research context.
		\item Key factors include sample size, effect size, risk tolerance, test type, and number of hypothesis tests.
		\item These factors influence statistical power and error probabilities in hypothesis testing.
	\end{itemize}
	
	\footnotesize \cite{Cohen1988,Baker2016,Herzog2019}
	
}

\frame{\frametitle{Research Question}
	
	\centering
	
	Can a data-driven and context-sensitive approach improve the choice of the significance level compared to the conventional value $\alpha = 0.05$?
	
}

\begin{frame}{Optimal Alpha Definition}
	
	\begin{columns}[T]
		
		\column{0.48\textwidth}
		\begin{block}{Replicated}
			Keep original $\alpha$  
			
			(decision validated)
		\end{block}
		
		\column{0.48\textwidth}
		\begin{alertblock}{Failed}
			Adjust $\alpha$: largest  
			
			$\alpha < p$-value  
			
			(correct decision)
		\end{alertblock}
		
	\end{columns}
	
	\vspace{0.5cm}
	
	\begin{exampleblock}{Example}
		
		$p = 0.047 \rightarrow$ failed to replicate  
		
		Optimal $\alpha = 0.04$ (not 0.05)
		
	\end{exampleblock}
	
	\vspace{0.2cm}
	
	\begin{block}{Features Used in the Model}
		
		\begin{itemize}
			\item Sample size
			\item Effect size
			\item Optimal Alpha (Target Variable)
		\end{itemize}
		
	\end{block}
	
\end{frame}

\section{Stage 2}
\begin{frame}{Stage 2: Context-Based Adjustment}
	
	\begin{columns}[T]
		
		\column{0.32\textwidth}
		\begin{block}{$k_1$ — Risk Tolerance}
			
			\begin{itemize}
				\item Very Strict $\times 0.50$
				\item Strict $\times 0.75$
				\item Neutral $\times 1.00$
				\item Lenient $\times 1.25$
				\item Very Lenient $\times 1.50$
			\end{itemize}
			
		\end{block}
		
		
		\column{0.32\textwidth}
		\begin{block}{$k_2$ — Test Direction}
			
			\begin{itemize}
				\item Two-tailed $\times 1.00$
				\item One-tailed $\times 2.00$
			\end{itemize}
			
			\vspace{0.3cm}
			
			Reflects rejection region allocation.
			
		\end{block}
		
		
		\column{0.32\textwidth}
		\begin{block}{$k_3$ — Multiple Testing}
			
			Bonferroni correction:
			
			\[
			\alpha / m
			\]
			
			where $m$ = number of hypothesis tests.
			
		\end{block}
		
	\end{columns}
	
	\vspace{0.4cm}
	
	\centering
	\small
	Final clamping: $0.001 \leq \alpha_{final} \leq 0.15$
	
\end{frame}
\begin{frame}{Final Alpha Calculation}
	
	\begin{block}{Adjusted Significance Level}
		
		\[
		\alpha_{final}
		=
		\alpha_{baseline}
		\times
		k_1(\text{risk})
		\times
		k_2(\text{direction})
		\times
		k_3(\text{multiple testing})
		\]
		
	\end{block}
	
	\vspace{0.5cm}
	
	\begin{exampleblock}{Interpretation}
		
		\begin{itemize}
			\item $\alpha_{baseline}$: predicted by the machine learning model.
			\item $k_1$: adjustment for risk tolerance.
			\item $k_2$: adjustment for test direction.
			\item $k_3$: adjustment for multiple hypothesis testing.
		\end{itemize}
		
	\end{exampleblock}
	
\end{frame}

\section{Limitations}
\begin{frame}{Limitations}
	
	\begin{block}{Study Limitations}
		
		\begin{itemize}
			\item Dataset derived from a limited number of replication studies.
			\item Study characteristics used as inputs are restricted to a few key variables.
			\item Model predictions depend on patterns observed in the available data.
		\end{itemize}
		
	\end{block}
	
	\vspace{0.4cm}
	
	\begin{exampleblock}{Implication}
		
		The recommended significance levels should be interpreted as guidance rather than a universal replacement for fixed thresholds.
		
	\end{exampleblock}

\section{Conclusion}
\begin{frame}{Conclusion}
	
	\begin{block}{Key Takeaways}
